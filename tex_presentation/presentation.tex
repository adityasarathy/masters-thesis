%% FRAME
\begin{frame}[fragile]{CTC loss}
	\alert{Idea}: decode sequence from fixed-width output
	\onslide<1->{
		\begin{equation}
			\begin{gathered}
			\label{eq:multiple}
			ACT = \begin{cases}
			decode(A, A, A, C, T) \\
			decode(A, A, C, -, T) \\
			decode(-, A, C, T, T)  \\
			decode(-, -, A, C, T)  \\
			decode(A, C, C, C, T)  \\
			\vdots \\
			decode(A, C, T, -, -) 
			\end{cases}
			\end{gathered}
		\end{equation}
	}
	\onslide<2->{	
		\begin{equation}
			\begin{gathered}
			P(Y | X) = \sum_{\pi \in decode^{-1}(Y)}^{} P(\pi | X)
			\end{gathered}
		\end{equation}
	}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% FRAME
\begin{frame}[fragile]{CTC loss}

Given the dataset $D = \{(X_i, Y_i)\}$, training objective is the maximization of the likelihood of each training sample  which is the same as the minimization of negative log likelihood:

	\begin{equation}
		\begin{gathered}
		L(D) = - \sum_{(X,Y)\in D}^{} ln P(Y | X)
		\end{gathered}
	\end{equation}
\end{frame}
